190 V finite_difference_hvp(Func &g, const V &weights, const V &input, const V &target, const V &v, double epsilon = 1e-6) {
191  V w_plus = weights + epsilon * v;

216 V lbfgs_two_loop(const std::vector<V>& s_list, const std::vector<V>& y_list, const std::vector<double>& rho_list, const V& v) { //two loop recursion to compute H*v
217  int M = s_list.size();
218  std::vector<double> alpha(M);
219  V q = v;
220 
221  // backward loop
222  for (int i = M - 1; i >= 0; --i) {
223  alpha[i] = rho_list[i] * s_list[i].dot(q);
224  q = q - alpha[i] * y_list[i];
225  }
226 
227  // Scaling
228  double gamma = 1.0;
229  if (M > 0) {
230  double denom = y_list.back().dot(y_list.back());
231  if (std::abs(denom) < 1e-12) {
232  gamma = 1.0;
233  } else {
234  gamma = s_list.back().dot(y_list.back()) / denom;
235  }
236  gamma = std::min(std::max(gamma, 1e-6), 1e6);
237  }
238  V r = gamma * q;
239 
240  // fowrard loop
241  for (int i = 0; i < M; ++i) {
242  double beta = rho_list[i] * y_list[i].dot(r);
243  r = r + s_list[i] * (alpha[i] - beta);
244  }
245  return r;
246 }
